#create middleware which will be run before any app.get() by using xxx() in convention would be next()

example:

    const setDate = function (req,res,next){
        req.date = Date.now()
        next()
    }

    app.use(setDate)

    app.get("",(req,res)=>{
        console.log(req.date)
        res.send(req.date)
    })
